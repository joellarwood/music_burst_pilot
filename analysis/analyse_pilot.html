pilot <- here::here(
  "data",
  "processed_pilot.csv"
) %>% 
  read_csv() %>% 
  mutate(
    song = gsub(
      "audio/",
      "",
      stimulus_song
    )
  ) %>% 
  rename(
    target = target_song
  )

In this study we sampled 30 participants, with each participant listening to 42 1 second exceprts. Participants rated each song in terms of the emotion they thought it expressed, its valence, and its arousal.

Considering Happy, Sad, and Angry taget songs

target_happy <- pilot %>% 
  select(target, song, correct, valence, arousal) %>% 
  filter(target == "happy") %>% 
  group_by(
    song
  ) %>% summarise(
    correct = sum(correct)/n(),
    valence = mean(valence),
    arousal = mean(arousal)
  ) %>% 
  arrange(correct)

selected_songs <- pilot %>% 
  select(target, song, correct, valence, arousal) %>% 
  group_by(song) %>%
  summarise(
    valence = mean(valence),
    arousal = mean(arousal),

And here the songs can be seen plotted in terms of valence and arousal

pilot %>% 
  select(target, song, correct, valence, arousal) %>% 
  group_by(song) %>%
  summarise(
    valence = mean(valence),
    arousal = mean(arousal),
    target = target,
    correct = sum(correct)/n()
  ) %>% 
  filter(
    song %in% c("25_Old_Blacktop.mp3", "12_Fire_And_Rain_(LP_Version).mp3", "15_Here_Today.mp3", "2_Love's_Requiem.mp3", "2_Inside.mp3", "12_Skank_Down.mp3", "26_Make_Light.mp3", "28_Valerie.mp3", "11_Be_Happy.mp3", "16_Restless.mp3", "13_Motown_Junk.mp3", "2_Incomplete.mp3")
  ) %>% 
  distinct() %>% 
  ggplot2::ggplot(
    aes(
      x = valence,
      y = arousal,
      shape = target,
      label = song,
      color = correct
    ) 
  )+
  geom_point() +
  theme_classic() +
  ggrepel::geom_text_repel() +
  coord_cartesian(xlim = c(1, 7), ylim = c(1, 7)) +
  ggplot2::ggtitle("Selected Songs plotted on Valence and Arousal") +
  scale_x_continuous(breaks = c(1:7)) +
  scale_y_continuous(breaks = c(1:7))
